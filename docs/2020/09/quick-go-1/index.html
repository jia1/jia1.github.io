package main

import "fmt"

func main() {

 fmt.Println(len("Hello, world!"))
 fmt.Println("Hello, world!"[1]) // prints 101, which represents 5 in binary, instead of 'e'
 fmt.Println("Hello, " + "world!")
 fmt.Println(true && !(true || false))
}
